# Compiler and flags
CXX = g++
CXXFLAGS = -std=gnu++17 -O2 -Wall -Wextra -Wno-unused-parameter

# Target executable
TARGET = myshell

# Source files
SRCS = command.cc tokenizer.cc

# Object files
OBJS = $(SRCS:.cc=.o)

# Default: build the shell
all: $(TARGET)

# Link object files into the final executable
$(TARGET): $(OBJS)
	$(CXX) $(CXXFLAGS) -o $@ $^

# Compile each .cc into .o
%.o: %.cc %.h
	$(CXX) $(CXXFLAGS) -c $< -o $@

# Clean builds
clean:
	rm -f $(OBJS) $(TARGET)

# Run the shell
run: $(TARGET)
	./$(TARGET)
